class Solution {
public:
    vector<int> replaceElements(vector<int>& arr) {
        if(arr.size() == 0)
            return {};
        
        int maxRight = arr[arr.size()-1];
        arr[arr.size()-1] = -1;
        for(int i = arr.size()-2; i >= 0; i --){
            int tmp = arr[i];
            
            arr[i] = maxRight;
            if(tmp > maxRight)
                maxRight = tmp;
        }
        return arr;
    }
};
